    ```typescript
    const chatPlaceholder = (() => {
      if (!activeConversation) return undefined
      const parsed = parseConversationId(activeConversation)
      if (parsed.type === 'department') {
        const dept = departments.find((d) => d.id === parsed.id)
        return dept ? `Message ${dept.name} department...` : undefined
      }
      if (parsed.type === 'team') {
        const team = teams.find((t) => t.id === parsed.id)
        return team ? `Message ${team.name} team...` : undefined
      }
      if (parsed.type === 'agent') {
        return `Message ${parsed.name}...`
      }
      return undefined
    })()
    ```

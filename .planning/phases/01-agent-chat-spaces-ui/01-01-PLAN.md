    1. **Types:**
       ```typescript
       export type ChatSpace = 'departments' | 'teams' | 'agents'

       export interface DepartmentChatEntity {
         conversationId: string   // `dept:${dept.id}`
         departmentId: number
         name: string
         color?: string
         leadAgent: Agent | null
         leadAgentId: number | null
       }

       export interface TeamChatEntity {
         conversationId: string   // `team:${team.id}`
         teamId: number
         name: string
         color?: string
         departmentName: string | undefined
         leadAgent: Agent | null
       }

       export interface ParsedConversationId =
         | { type: 'department'; id: number }
         | { type: 'team'; id: number }
         | { type: 'agent'; name: string }
         | { type: 'unknown' }
       ```

    2. **getDepartmentChatEntities(departments: Department[], agents: Agent[]): DepartmentChatEntity[]**
       - Map each department to a DepartmentChatEntity
       - Resolve leadAgent by finding `agents.find(a => a.id === dept.manager_agent_id)` — return null if not found or manager_agent_id undefined
       - Set conversationId to `dept:${dept.id}`

    3. **getTeamChatEntities(teams: Team[], departments: Department[], assignments: AgentTeamAssignment[], agents: Agent[]): TeamChatEntity[]**
       - For each team, find lead via `assignments.find(a => a.team_id === team.id && a.role === 'lead')`
       - Resolve leadAgent from agents array using lead assignment's agent_id — null if no lead or agent not found
       - Resolve departmentName via `departments.find(d => d.id === team.department_id)?.name`
       - Set conversationId to `team:${team.id}`

    4. **parseConversationId(id: string): ParsedConversationId**
       - `dept:N` -> `{type:'department', id: Number(N)}`
       - `team:N` -> `{type:'team', id: Number(N)}`
       - `agent_NAME` -> `{type:'agent', name: NAME}`
       - anything else -> `{type:'unknown'}`

    5. **getSystemMessageForStatus(agentName: string, status: Agent['status']): string | null**
       - busy: `"${agentName} is busy. Your message is queued and will be delivered."`
       - offline: `"${agentName} is offline. Your message will be delivered when available."`
       - error: `"${agentName} is in an error state. Your message was sent but response may be delayed."`
       - idle: return null

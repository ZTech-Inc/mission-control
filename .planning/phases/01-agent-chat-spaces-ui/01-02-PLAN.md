       ```typescript
       const pendingIdRef = useRef(-1)
       const [isGenerating, setIsGenerating] = useState(false)

       async function handleSend(content: string, attachments?: ChatAttachment[]) {
         pendingIdRef.current -= 1
         const tempId = pendingIdRef.current
         const optimistic: ChatMessage = {
           id: tempId,
           conversation_id: conversationId,
           from_agent: 'human',
           to_agent: targetAgentName,
           content,
           message_type: 'text',
           attachments,
           created_at: Math.floor(Date.now() / 1000),
           pendingStatus: 'sending',
         }

         setMessages(prev => [...prev, optimistic])
         setIsGenerating(true)

         try {
           const res = await fetch('/api/chat/messages', {
             method: 'POST',
             headers: { 'Content-Type': 'application/json' },
             body: JSON.stringify({
               from: 'human',
               to: targetAgentName,
               content,
               conversation_id: conversationId,
               message_type: 'text',
               attachments,
               forward: true,
             }),
           })

           if (res.ok) {
             const data = await res.json()
             if (data.message) {
               setMessages(prev => prev.map(m => m.id === tempId ? data.message : m))
             }
             // Add system message for non-idle agent
             if (targetAgentStatus !== 'idle') {
               const sysMsg = getSystemMessage(targetAgentName, targetAgentStatus)
               if (sysMsg) {
                 setMessages(prev => [...prev, {
                   id: Date.now(),
                   conversation_id: conversationId,
                   from_agent: 'system',
                   to_agent: null,
                   content: sysMsg,
                   message_type: 'status',
                   created_at: Math.floor(Date.now() / 1000),
                 }])
               }
             }
           } else {
             setMessages(prev => prev.map(m => m.id === tempId ? { ...m, pendingStatus: 'failed' } : m))
           }
         } catch {
           setMessages(prev => prev.map(m => m.id === tempId ? { ...m, pendingStatus: 'failed' } : m))
         } finally {
           setIsGenerating(false)
         }
       }
       ```

       ```typescript
       function getSystemMessage(name: string, status: Agent['status']): string | null {
         switch (status) {

           case 'offline': return `${name} is offline. Your message will be delivered when available.`
           case 'error': return `${name} is in an error state. Your message was sent but response may be delayed.`
           default: return null
         }
       }
       ```

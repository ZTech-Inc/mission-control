**Create new file `src/app/api/agents/create/route.ts`:**

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { mkdirSync, writeFileSync } from 'node:fs'
import path from 'node:path'
import { createHash } from 'node:crypto'
import { z } from 'zod'
import { requireRole } from '@/lib/auth'
import { config } from '@/lib/config'
import { getDatabase } from '@/lib/db'
import { invalidateOrgSnapshot } from '@/lib/org-scanner'
```

1. Define zod schema for the body:
   ```
   const CreateAgentBody = z.object({
     name: z.string().min(1).max(100),
     role: z.string().optional().default(''),
     model: z.string().optional().default(''),
     identity_md: z.string().optional().default(''),
     agent_md: z.string().optional().default(''),
     soul_md: z.string().optional().default(''),
     department_name: z.string().min(1),
     team_name: z.string().optional(), // omit for department manager (MANAGER/ dir)
     is_manager: z.boolean().optional().default(false),
     tool_allow: z.array(z.string()).optional().default([]),
     tool_deny: z.array(z.string()).optional().default([]),
     tool_profile: z.string().optional().default(''),
     model_primary: z.string().optional().default(''),
     model_fallback: z.array(z.string()).optional().default([]),
   })
   ```
2. Guard: if `!config.agentsDir` return 400 `{ error: "Agents directory not configured" }`.
3. Compute agent directory:
   - If `is_manager` is true: `agentDir = path.join(config.agentsDir, body.department_name, 'MANAGER')`
   - Else: `agentDir = path.join(config.agentsDir, body.department_name, body.team_name, body.name)`
4. `mkdirSync(agentDir, { recursive: true })`
5. Write files:
   - `writeFileSync(path.join(agentDir, 'IDENTITY.md'), body.identity_md || '# ' + body.name + '\n\n' + (body.role ? 'Role: ' + body.role : ''), 'utf-8')`
   - `writeFileSync(path.join(agentDir, 'AGENT.md'), body.agent_md || '# Agent Configuration\n\nName: ' + body.name, 'utf-8')`
   - `writeFileSync(path.join(agentDir, 'SOUL.md'), body.soul_md || '# Soul\n\nPurpose: ' + (body.role || 'General agent'), 'utf-8')`
6. Compute `externalId` using stableNumber with key `'agent:' + agentDir`.
7. INSERT into `agents` table with `source = 'manual'`, setting name, role, model, workspace_path = agentDir.
8. If NOT a manager and `team_name` is provided, also INSERT into `agent_team_assignments` with `source = 'manual'`.
9. If IS a manager, UPDATE the department record: `db.prepare('UPDATE departments SET manager_agent_id = ? WHERE workspace_id = ? AND name = ?').run(agentId, workspaceId, body.department_name)`.
10. `invalidateOrgSnapshot(workspaceId)`
11. Return 201 `{ ok: true, agent: { id, name, role } }`.

  ```typescript
  import { parseField, parseListField, type ParsedAgentMetadata } from './org-scanner'

  export interface ParsedAgentProfile extends ParsedAgentMetadata {
    protocol_stack: string[]
    deliverables: string[]
    dependencies: string[]
    preferred_runtime: string | undefined
    openclaw_id: string
  }

  export function parseAgentProfile(
    agentDirName: string,
    agentMd: string,
    identityMd: string
  ): ParsedAgentProfile {
    // Import existing parseAgentMetadata logic by calling the parse utilities directly
    // (do NOT import parseAgentMetadata — it is internal and calls the same utilities)

    // Parse name using parseField with keys ['name'] from both sources
    // Parse role using parseField with keys ['role', 'title'] from both sources
    // Parse skills using parseListField with keys ['skills', 'capabilities'] from both sources
    // Parse kpis using parseListField with keys ['kpis', 'goals', 'metrics'] from both sources
    // Parse department and team using parseField
    // Parse assignmentRole using parseField with keys ['assignment_role', 'team_role', 'org_role']

    // NEW fields:
    const protocol_stack = [
      ...parseListField(agentMd, ['protocol_stack', 'protocols', 'protocol stack']),
      ...parseListField(identityMd, ['protocol_stack', 'protocols', 'protocol stack']),
    ]
    const deliverables = [
      ...parseListField(agentMd, ['deliverables', 'outputs']),
      ...parseListField(identityMd, ['deliverables', 'outputs']),
    ]
    const dependencies_parsed = [
      ...parseListField(agentMd, ['dependencies', 'depends_on', 'depends on']),
      ...parseListField(identityMd, ['dependencies', 'depends_on', 'depends on']),
    ]
    const preferred_runtime =
      parseField(agentMd, ['preferred_runtime', 'runtime']) ||
      parseField(identityMd, ['preferred_runtime', 'runtime'])

    // Derive openclawId using EXACT formula from agent-workspace.ts line 45:
    const rawName = name || agentDirName
    const openclaw_id = rawName.toLowerCase().replace(/[^a-z0-9._-]+/g, '-')

    return {
      name: name || agentDirName,
      role,
      skills: [...new Set(skills)],
      kpis: [...new Set(kpis)],
      department,
      team,
      assignmentRole,
      protocol_stack: [...new Set(protocol_stack)],
      deliverables: [...new Set(deliverables)],
      dependencies: [...new Set(dependencies_parsed)],
      preferred_runtime,
      openclaw_id,
    }
  }
  ```
